CREATE DATABASE `spring_bazaar` /*!40100 DEFAULT CHARACTER SET utf8 */;
DROP TABLE IF EXISTS `spring_bazaar`.`sb_users`;
CREATE TABLE `spring_bazaar`.`sb_users` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `login` VARCHAR(45) NOT NULL,
  `password` VARCHAR(45) NULL,
  `role_id` INT NULL,
  `person_id` INT NULL,
  PRIMARY KEY (`id`));
  
  DROP TABLE IF EXISTS `spring_bazaar`.`sb_roles`;
  CREATE TABLE `spring_bazaar`.`sb_roles` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(45) NOT NULL,
  `user_id` INT NULL,
  PRIMARY KEY (`id`));

  DROP TABLE IF EXISTS `spring_bazaar`.`sb_persons`;
  CREATE TABLE `spring_bazaar`.`sb_persons` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `user_id` INT NULL,
  `first_name` VARCHAR(45) NULL,
  `middle_name` VARCHAR(45) NULL,
  `last_name` VARCHAR(45) NULL,
  `product_id` INT NULL,
  PRIMARY KEY (`id`));
  
  DROP TABLE IF EXISTS `spring_bazaar`.`sb_products`;
  CREATE TABLE `spring_bazaar`.`sb_products` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `description` varchar(255) DEFAULT NULL,
  `price` decimal(10,2) DEFAULT NULL,
  `image_url` varchar(200) DEFAULT NULL,
  `person_id` int(11) NOT NULL,
  PRIMARY KEY (`id`)) ;
